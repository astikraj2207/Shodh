//FAST IO//
void FAST() {
	ios::sync_with_stdio(0);
	cin.tie(0);
	cout.tie(0);
}

vector <int> adj[100001];
bool visited[100001];

void addEdge(int a, int b)
{
    adj[a].pb(b);
    adj[b].pb(a);
}

void dfs(int i)
{
  if (visited[i]==true) {return;}
  
visited[i]=true;

for (auto c : adj[i])
	{
	dfs(c);
	}
}

/********************************************************************/
int main() {
	FAST();

	CASE(t) {
    for (int i=0; i<100001; i++)
    {
        adj[i].clear();
        visited[i]=0;
    }
    
    int n,m; cin>>n>>m;
    for (int i=0; i<m; i++)
    {
        int a,b; cin>>a>>b;
        addEdge(a,b);
    }
    
    int p=0, count=0; int prod=1;
    for (int i=1; i<n+1; i++)
    {
        if (visited[i]==false)
        {
            dfs(i,&p);
            count++;
            prod *=p;
            p=0;
        }
    }
    cout<<count<<" "<<prod<<endl;
    
	}

	return 0;
}
